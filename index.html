function selectionSort(arr) {
  let n = arr.length;
  for (let i = 0; i < n - 1; i++) {
    let min_idx = i;
    for (let j = i + 1; j < n; j++) {
      if (arr[j] < arr[min_idx]) {
        min_idx = j;
      }
    }
    if (min_idx !== i) {
      swap(arr, i, min_idx);
    }
  }
}

function bubbleSortDescending(arr) {
  let n = arr.length;
  for (let i = 0; i < n - 1; i++) {
    let swapped = false;
    for (let j = 0; j < n - i - 1; j++) {
      if (arr[j] < arr[j + 1]) {
        swap(arr, j, j + 1);
        swapped = true;
      }
    }
    if (!swapped) break;
  }
}

function insertionSort(arr) {
  let n = arr.length;
  for (let i = 1; i < n; i++) {
    let key = arr[i];
    let j = i - 1;
    while (j >= 0 && arr[j] > key) {
      arr[j + 1] = arr[j];
      j = j - 1;
    }
    arr[j + 1] = key;
  }
}

Mengapa Diberi Pengujian Empiris?

Perhitungan kompleksitas teori Big-O ($\mathcal{O}(n^2)$) sering kali terasa abstrak. Dengan menjalankan komparasi eksekusi data acak tanpa keterlambatan animasi UI, kita dapat membuktikan performa sebenarnya dari rancangan loop bertingkat (*nested loop*) di mesin komputasi browser Anda.

Peringatan Kinerja:

Dataset di atas 3.000 elemen pada Bubble Sort membutuhkan operasi iterasi yang tinggi. Hal ini membuktikan kelemahan fatal algoritma kuadratik!

Kesimpulan Pembahasan

Rangkuman akhir serta panduan dalam memilih algoritma sorting.

Pilih Selection Sort Jika...

Menulis data ke memori fisik dinilai sangat mahal/berat, karena algoritma ini menjamin hanya ada maksimal $\mathcal{O}(n)$ kali proses pertukaran (swap) nilai selama pengerjaan.

Pilih Bubble Sort Jika...

Data berukuran sangat kecil atau hanya digunakan sebagai media pembelajaran visual dasar bagi pemula, mengingat beban algoritmanya yang berat dan lambat pada data dalam jumlah besar.

Pilih Insertion Sort Jika...

Kumpulan data yang diolah sudah hampir terurut, atau jika Anda berurusan dengan aliran data real-time (online algorithm) karena ia mampu mengurutkan data baru secara instan dan efisien.
